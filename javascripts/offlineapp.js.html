var parseForm = function(workorder) {
	var values = {};
	var data = $('#' + workorder).serializeArray();
 
	var mapValues = function(index, field) {
		values[field.name] = field.value;
	}
 
	$(data).each(mapValues);
	
	return values;
}
 
var saveLocally = function(data) {
	var workOrderCount = parseInt(localStorage.workOrderCount) + 1 || 1;
  
	localStorage['workorder' + workOrderCount] = data;
	localStorage.workOrderCount = workOrderCount;
}
 
var sendLocalStorageToServer = function() {
	for(key in localStorage) {
		if(key.indexOf("workorder") > -1) {
			checkConnected(localStorage.getItem(key));
			localStorage.removeItem(key);
			localStorage.workOrderCount--;
		}
	}
}
 
var checkConnected = function(values, connectedCallback, notConnectedCallback) {
	var xhr = new XMLHttpRequest();
	var handler = function() {
		if(xhr.readyState === 4) {
			if(xhr.status === 200) {
				if(connectedCallback !== undefined)
					connectedCallback();
				changeNotification('Work Order Successfully Sent to Server', 'rgba(15, 150, 65, 1)');
				setTimeout(function() { changeNotification('&nbsp', 'white') }, 5000);
			}
			else {
				notConnectedCallback(values);
				changeNotification('No Connection -- Work Order Stored Locally', 'rgba(205, 0, 0, 1)');
				setTimeout(function() { changeNotification('&nbsp', 'white') }, 5000);
			}
		}
	}
	
	xhr.onreadystatechange = handler;
	xhr.open("POST", "http://localhost:8080", true);
	xhr.send(values);
}

var validate = function() {
	var isValid = true;
	var elements = ['firstname', 'lastname', 'theDate', 'theDescription'];
 
	for(var i = 0; i < elements.length; i++) {
		if(document.getElementById(elements[i]).value.length < 1) {
			document.getElementById(elements[i]).style.borderColor = 'rgba(205, 0, 0, 1)';
			document.getElementById(elements[i]).style.borderWidth = '2px';
			changeNotification('Please provide values for the highlighted field(s).', 'rgba(205, 0, 0, 1)');
			isValid = false;
		}
	}
 
	return isValid;
}
